public static PrepareInsertsOutput prepareInserts(PrepareInsertsInput input) {
    PrepareInsertsOutput output = new PrepareInsertsOutput();
    output.setName(input.getName());

    output.setTime(Timestamp.from(input.getTime()));
    return output;
